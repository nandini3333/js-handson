            Note: You have to complete Game_Winner function. No need to take any input.
     

     
         sol : 
                   var Game_Winner = (S) => 
                   {
                         let countA = S.match(/[A]/gi).length;
                         let countB = S.match(/[D]/gi).length;
                         if(countA > countB){
                                return "Aditya"
                         }
            
                        else if(countA < countB){

        9) Reverse the String
            You are given a string S, Your task is to Reverse the string. String can contain both 
            upppercase lowercase letters. 
            Note: You have to complete Reverse_String function. No need to take any input.
     

    
        sol :
                  var Reverse_String = (str) => 
                  {
                          let b = str.split('');
                          let c = b.reverse();
                          return c.join('');
                 };
