        2) Create one function which takes two values as a parameter and print the sum 
            of them as "Sum of 3, 4 is 7"
    
        sol : 
              function myFunction(a,b){
                    let sum = a + b;
                console.log(" sum of " + a + "," + b + " is " + sum);
              }
        
              myFunction(3,4);

        7) Write a function that accepts parameter n and returns factorial of n

       sol : 
             var n = 4;
             var fact = 1;
             for( let i = 1; i <= n ;i++){
                fact = fact * i;
             }
             console.log(fact);
